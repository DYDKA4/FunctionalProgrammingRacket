(define (mySum L)
      (apply + L))
(define (myMul L)
      (apply * L))
(define (myGeom arr)
    (expt (abs(myMul arr)) (/ 1 (length arr))
          )
    )
(define (build_arr arr)
    (map (lambda (list)
           (expt (myGeom list) 2))
         arr
         )
   )
(define (task-03 lst)
    (expt (mySum (build_arr lst)) 1/2)
    )
